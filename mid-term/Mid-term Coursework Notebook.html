"""
    This function creates a 4x4 grid of subplots to visualize the interactions between the features in the input data.
    Parameters:
    - X : 2D array of feature data
    - y : 1D array of class labels corresponding to the data in X
    Returns:
    - None
"""    
def myplotGrid(X,y):
    # Initialize the figure and axes
    fig, axes = plt.subplots(4, 4, figsize=(12, 12))
    fig.suptitle("Features Interaction Visualization")
    
    # Create a list of arrays for each class
    dataPerClass = [np.array([X[n] for n in range(0, len(X)) if y[n] == i]) for i in range(3)]

    # Loop over the subplots
    for i in range(4):
        for k in range(4):
            ax = axes[i][k]
            
            # Plot histograms for diagonal subplots
            if i == k:
                for d in dataPerClass:
                    ax.hist(d[:, i])
            # Plot scatter plots for non-diagonal subplots
            else:
                for l, d in enumerate(dataPerClass):
                    ax.scatter(d[:, i], d[:, k], label=wine.target_names[l])
            
            # Add legend in top right corner
            if i == 3 and k == 0:
                ax.legend()
            
            # Add x-label for bottom row of subplots
            if k == 3:
                ax.set_xlabel(wine.feature_names[i])
            
            # Add y-label for leftmost column of subplots
            if i == 0:
                ax.set_ylabel(wine.feature_names[k])

        # Assert that the length of arrays j and k are the same

'''Returns predicted values when given training data and labels, a test set, k neighbours and distance type.
        -----------
        Takes:
           - X=training inputs value, 
           - y=training outputs value, 
           - X=test inputs value, 
           - k=number of neighbours, 
           - distance_type= "euclidian" or "manhattan"
        Returns:
           - List contains result of predicted values for those test inputs.
        -----------
'''
def mykNN(X_train, y_train, X_test, k, distance_type):
    predicted=[]
# Every test sample's fit and prediction
    for x in X_test:
        neighbours=[[],[]]
    # Get the euclidian or manhattan distances between all of the training samples and the test sample.
        if distance_type=="euclidean":
            distances = [euclidean_dist(x, j) for j in X_train]
        elif distance_type=="manhattan":
            distances = [manhattan_dist(x, j) for j in X_train]
    # Get indices of the k smallest distances
        k_nearest_ids = np.argsort(distances)[:k]
    # Collect the k nearest neighbors' labels.
        k_nearest_labels = [y_train[m] for m in k_nearest_ids]
        counts = np.bincount(k_nearest_labels)
    # Among the k closest neighbors, predicted class is the most prevalent class.
        predicted.append(np.argmax(counts))
    return np.array(predicted)

#the method should return the confusion matrix in a numpy array
''' Returns a confusion matrix of predicted and ground truth values.
        ----------
        Takes:
            - Array of test data labels,y_test
            - Array of predicted values,y_pred
            - Number of classes
        Returns:
            - nxn Confusion matrix
        ----------
'''
def Confusion_Matrix(y_test,y_pred,nClasses):
    # Initialise an n dimensional matrix of zeros
    e = np.zeros((nClasses,nClasses),dtype=np.int64)
    # Loop over predicted values updating matrix appropriately
    for i in range(0, len(y_pred)):
        e[y_test[i]][y_pred[i]] +=1
    #loop through all results and update the confusion matrix
    return e

''' Gets class by class precision 
        ----------
        Takes:
            - Test outputs,y_test
            - Predicted values,y_pred
        Returns:
            - List with precision for each class  
        ----------
'''
def Precision(y_test,y_pred):
    # Get array of classes
    classes = np.unique(y_test)
    # Get number of classes
    n = len(classes)
    # Initialise array of zeros for class by class results
    precision = np.zeros(classes.shape)
    # get confusion matrix
    cm = Confusion_Matrix(y_test, y_pred, n)
    # loop over results and classes
    for i in range(0, n):
        # Initialise true positive + false positive variable
        sum_of_tpfp = 0
        for j in classes:
            # true positives + false positives
            sum_of_tpfp += cm[j][i]
        # precision = true positives / true positives + false positives
        class_precision = cm[i][i] / sum_of_tpfp
        precision[i] = class_precision
    return precision

''' Gets class by class Recall 
        ----------
        Takes:
            - Test outputs,y_test
            - Predicted values,y_pred
        Returns:
            - List contains of with recall for each class
        ----------
'''
def Recall(y_test,y_pred):
       # Get array of classes
    classes = np.unique(y_pred)
    # Get number of classes
    n = len(classes)
    # Initialise array of zeros for class by class results
    recall = np.zeros(classes.shape)
    # Get confusion matrix
    cm = Confusion_Matrix(y_test, y_pred, n)
    # Loop over the classes
    for m in classes:
        # Recall = true positives / (true positives + false negatives)
        recall[m] = cm[m][m] / sum(cm[m])
    return recall

#use the numpy function where to return the accuracy given the true/predicted labels.
'''Calculates overall accuracy
        ----------
            Takes:
                - list of test values,y_test
                - list of predicted values,y_pred
            Returns:
                - Accuracy value
        ----------
    '''
def Accuracy(y_test,y_pred):
    accuracy = len(y_test[np.where(y_test==y_pred)])/len(y_pred) #np.where()
    return accuracy

# NestedCrossVal code
""" Nest Cross Validation of Knn.
        ----------
        Takes: 
            - inputs, 
            - outputs, 
            - number of folds, 
            - list: range of neighbour values, 
            - list of distance names: "euclidean","manhattan"
            - random seed.
        Returns: 
            - list of accuracies, 
            - list of parameters, 
            - list of confusion matrices
        ----------
    """
def Nested_CrossVal(X,y,foldk,nns,dists,mySeed):
    #conf_matrices, accuracy_fold, parameters_fold are initialised
    np.random.seed(mySeed)
    conf_matrices=[]
    accuracy_fold=[]
    parameters_fold=[]
    #TASK: use the function np.random.permutation to generate a list of shuffled indices from in the range (0,number of data)  
    
    #list of shuffled indices are generated
    indices = np.random.permutation(list(range(X.shape[0])))
    
    #TASK: use the function array_split to split the indices to foldK different bins
    #for splitting the indices to different bins array_split is used
    bins=np.array_split(indices, foldk)
    
    #proving foldK and bin num length values are equal
    assert(foldk==len(bins))
    #foldK is looped
    for i in range(foldk):
# indices for testing
        foldTest = bins[i]
# indices for validation
        foldVal = bins[(i+1)%foldk]
# indices for training
        foldTrain = [index for j, bin in enumerate(bins) if j not in [i, (i+1)%foldk] for index in bin]
        
        #save the best attained accuracy here (in terms of validation)
        bestAccuracy=-10
        #save the best number of neighbours here
        bestNumNeighbours=-1
        #save the best distance metric here
        bestDistance=''
        
        # loop through all parameters (one for loop for distances, one for loop for nn)
        # train the classifier on current number of neighbours/distance
        # obtain results on validation set
        #save parameters if results are the best we had
        for d in dists:
            #nns is looped
            for nn in nns:
                #knn classifier is defined
                y_pred=mykNN(X[foldTrain],y[foldTrain],X[foldVal],nn,d)
                #accuracy value is generated
                accuracy = Accuracy(y[foldVal], y_pred)
                #the accuracy values are compared
                if accuracy > bestAccuracy:
                    bestAccuracy = accuracy
                    bestNumNeighbours = nn
                    bestDistance = d
        
        #by including the validation values the training data set in extended
        foldTrain.extend(foldVal)
        
        #passing kNN on fold values
        y_=mykNN(X[foldTrain],y[foldTrain],X[foldTest],bestNumNeighbours,bestDistance)
        
        #Accuracy measurement using fold values
        accuracy_=Accuracy(y[foldTest],y_)
        #using .append() values are added into accuracy_value
        accuracy_fold.append(accuracy)
        parameters_fold.append((bestDistance, bestNumNeighbours))
        
        #no.of unique classes are stored
        nclasses=len(np.unique(y_))
        
        #confusion matrix is generated
        cm=Confusion_Matrix(y[foldTest],y_,nclasses)
        #updating conf_matrices
        conf_matrices.append(cm)

    return accuracy_fold,parameters_fold, np.array(conf_matrices);

"""
    This function calculates the precision for each class in the confusion matrix.
    precision = true positives / true positives + false positives
    
    Parameters:
    conf_matrix (list of lists): a confusion matrix
    
    Returns:
    list: a list of precision values for each class
"""
def Precision(conf_matrix):
    precision = []
    # Loop through each class
    for i in range(len(conf_matrix)):
        #true positives
        true_positives = conf_matrix[i][i]
        #false positives
        false_positives = sum([conf_matrix[j][i] for j in range(len(conf_matrix)) if j!=i])
        # true positives + false positives
        sum_of_tpfp = true_positives + false_positives
        # precision = true positives / true positives + false positives
        precision.append(true_positives / sum_of_tpfp)
    return precision

Q4. Tie break¶

Assume that you have selected the number of neighbours to be an even number, e.g., 2. For one of the neighbours, the suggested class is 1, and for the other neighbour the suggested class is 2. How would you break the tie? Write example pseudocode that does this.

Answer:¶

In this situation, we ought to pick the class whose values are most similar.

In [ ]:
neighbours=[]
# Assume that the list of neighbours is called neighbours and their corresponding classes are called classes

# Initialize variables to store the count of each class
count_class_1 = 0
count_class_2 = 0

# Loop through the neighbours
for i in range(len(neighbours)):
    # If the class of the current neighbour is 1, increment the count of class 1
    if classes[i] == 1:
        count_class_1 += 1
    # If the class of the current neighbour is 2, increment the count of class 2
    elif classes[i] == 2:
        count_class_2 += 1

# Compare the counts of each class
if count_class_1 > count_class_2:
    # Class 1 is the majority class, so break the tie in favor of class 1
    suggested_class = 1
else:
    # Class 2 is the majority class, so break the tie in favor of class 2
    suggested_class = 2

Q5. Improving performance on noisy data.¶

The performance of k-NN on the noisy data should be worse than on the clean data. Suggest at least one way of improving the performance on the noisy data. Try to elaborate on your idea as much as possible, including pseudocode where possible.

Answer:¶

The adoption of a weighting system makes it possible for each neighbor to have a weight that represents the likelihood (or probability) that it belongs to the class in addition to the distance between them. The class is determined by combining the weights and distances.
